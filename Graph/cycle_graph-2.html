Exemple jouet¶

Afin de faciliter l'criture des algorithmes et afin de vrifier leur pertinence nous avons gnr un jeu de donnes, pour lequel la recherche du nombre d'inversions est triviale. Cet exemple peut tre rsum tel que :

  • Organisme A : (+)A - (+)B - (+)C - (+)D - (+)E
  • Organisme B : (-)D - (-)C - (+)A - (+)B - (+)E

Le signe (+) implique que la position du dbut du gne est inffieure celle de la fin, et rciproquement pour le signe (-).

def find_edges(d, ref):
    """Cette fonction permet de chercher les arcs, soit les gnes adjacents sur un chromosome. Cette fonction prend en
    argument  la matrices des positions de dbut et de fin, de chaque segement orthologue. Cette matrice doit tre trie
    selon les positions de dbut. Pour chaque gne, la fonction `find_edge`, dtermine la 'fin' de l'arc comme tant la position
    maximale du gne courant (i) et dtermine le dbut de cette arc comme la position minimale du gne suivant (i+1). Enfin tant 
    donn que nous tudions des chromosomes bactriens, une arc doit tre cre entre la position minimale du premier gne et la
    position maximale du dernier gne."""
    
    # Dfinition des colonnes (d'aprs notre fonction les noms de colonnes de tableau sont fixs)
    if ref == True :
        start_column = "Qstart"
        end_column = "Qend"
        S = list(d["Qstart"])
        E = list(d["Qend"])
    else :
        start_column = "Sstart"
        end_column = "Send"
        S = list(d["Sstart"])
        E = list(d["Send"])
        
    l = []
    for i in range(len(E)):
        # Dfinition de la position de fin
        max_c = max(E[i], S[i])
        if max_c == E[i]:
            status_begin = "E"
        else: 
            status_begin = "S"
            
        if i < len(E)-1 :  
            # Dfinition de la position de dbut
            min_suiv = min(E[i+1], S[i+1])
            if min_suiv == E[i+1]:
                status_end = "E"
            else: 
                status_end = "S"
            
            # Recherche des IDs correspondant au maximum et au minimum
            # ID du dbut
            if status_begin == "E":
                begin_ID = list(d["ID"][d[end_column]==max_c] )[0] 
                name_beg = status_begin + str(begin_ID)
                
            else :
                begin_ID = list(d["ID"][d[start_column]==max_c] )[0]
                name_beg = status_begin + str(begin_ID)
                
                
            # ID de fin
            if status_end == "E":
                end_ID = list(d["ID"][d[end_column]==min_suiv] )[0] 
                name_end = status_end + str(end_ID)
                
            else :
                end_ID = list(d["ID"][d[start_column]==min_suiv] )[0]
                name_end = status_end + str(end_ID)
                
            # Ajout du rsultat
            l.append((name_beg, name_end))

        # Pour l'arc entre la position min et max
        else :
            # Position maximale
            max_last = max(E[-1],S[-1])
            if max_last == S[-1]:
                status_max_last = "S"
            else :
                status_max_last = "E"   
            min_first = min(E[0],S[0])
            
            # Position minimale
            if min_first == S[0]:
                status_min_first = "S"
            else :
                status_min_first = "E" 
            # Recherche des IDs 
            if status_max_last == "E":
                begin_ID = list(d["ID"][d[end_column]==max_last] )[0] 
                name_beg = status_max_last + str(begin_ID)
            else :
                begin_ID = list(d["ID"][d[start_column]==max_last] )[0]
                name_beg = status_max_last + str(begin_ID)
                
            if status_min_first == "E":
                begin_ID = list(d["ID"][d[end_column]==min_first] )[0] 
                name_end = status_min_first + str(begin_ID)
            else :
                begin_ID = list(d["ID"][d[start_column]==min_first] )[0]
                name_end = status_min_first + str(begin_ID)
                
             # Ajout du rsultat
            l.append((name_beg, name_end))
            
            
    return l

def scenari_aleatoire (QS, QE, SS, SE, inv_obs ,n):
    """ Cette fonction gnre des positions alatoires pour le gnome de refrence (Q), et pour le gnome d'intrt (S).
    Puis elle calcule le graph correspndant,  partir de celui-ci, elle infre le nombre d'inversions.
    Cette opration est rpte n fois. Enfin elle retourne la proportion de graphs alatoirement alatoirement
    gnrs qui permettent d'obtenir un nombre d'inversions infrieur  celui observ (inv_obs) """
    
    inv_alea= []
    for i in range(n):
        T1 = QS+QE
        random.shuffle(T1)
        QSshuffled = T1 [:int(len(T1)/2)]
        QEshuflled = T1[int(len(T1)/2) :]
        data_ref = {'Qstart': QSshuffled,'Qend': QEshuflled  }
        data_ref =  pd.DataFrame(data=data_ref)
        data_ref = data_ref.sort_values('Qstart')
        data_ref["ID"]= range(0, len(QSshuffled))
        
        T2 = SS+SE
        random.shuffle(T2)
        SSshuffled = T2 [:int(len(T2)/2)]
        SEshuflled = T2[int(len(T2)/2) :]
        data_int = {'Sstart': SSshuffled,'Send': SEshuflled  }
        data_int =  pd.DataFrame(data=data_int)
        data_int = data_int.sort_values('Sstart')
        data_int["ID"] = data_ref['ID']
            
     
        
        EDGES1 = find_edges(data_ref, True)
        EDGES2 = find_edges(data_int, False)
        
        G=nx.Graph()
        for i in EDGES1 :
            G.add_edge(i[0],i[1],color='r')
        for j in EDGES2 :
            G.add_edge(j[0],j[1],color='b')
        Composantes_connexes =[len(c) for c in sorted(nx.connected_components(G), key=len, reverse=True)]
        Nb_cycles = len(Composantes_connexes)
        Nb_inv = len(d_ref_jouet["Qstart"]) - Nb_cycles
        inv_alea.append(Nb_inv)
    c =0
    for i in range(n):
        if i < inv_obs:
            c +=1
    p = c/n
    return  p
